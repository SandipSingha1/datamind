USE ROLE ACCOUNTADMIN;

-- 1. Warehouse
CREATE WAREHOUSE IF NOT EXISTS DATAMIND_WH
  WAREHOUSE_SIZE      = 'XSMALL'
  AUTO_SUSPEND        = 60
  AUTO_RESUME         = TRUE
  INITIALLY_SUSPENDED = TRUE;

-- 2. Database + Schemas
CREATE DATABASE IF NOT EXISTS DATAMIND_DB;
CREATE SCHEMA  IF NOT EXISTS DATAMIND_DB.METADATA;
CREATE SCHEMA  IF NOT EXISTS DATAMIND_DB.RAW;

-- 3. Role
CREATE ROLE IF NOT EXISTS DATAMIND_ROLE;

-- 4. User  ← change this password
CREATE USER IF NOT EXISTS DATAMIND_USER
  PASSWORD             = 'DataMind@2026!'
  DEFAULT_WAREHOUSE    = DATAMIND_WH
  DEFAULT_NAMESPACE    = DATAMIND_DB.PUBLIC
  DEFAULT_ROLE         = DATAMIND_ROLE
  MUST_CHANGE_PASSWORD = FALSE;

-- 5. Assign role to user AND to SYSADMIN (so you can also use it)
GRANT ROLE DATAMIND_ROLE TO USER DATAMIND_USER;
GRANT ROLE DATAMIND_ROLE TO ROLE SYSADMIN;

-- 6. Warehouse
GRANT USAGE ON WAREHOUSE DATAMIND_WH TO ROLE DATAMIND_ROLE;

-- 7. Database
GRANT OWNERSHIP ON DATABASE DATAMIND_DB       TO ROLE DATAMIND_ROLE COPY CURRENT GRANTS;
GRANT OWNERSHIP ON SCHEMA DATAMIND_DB.METADATA TO ROLE DATAMIND_ROLE COPY CURRENT GRANTS;
GRANT OWNERSHIP ON SCHEMA DATAMIND_DB.RAW      TO ROLE DATAMIND_ROLE COPY CURRENT GRANTS;
GRANT OWNERSHIP ON SCHEMA DATAMIND_DB.PUBLIC   TO ROLE DATAMIND_ROLE COPY CURRENT GRANTS;

-- 8. ACCOUNT_USAGE access (for cost agent)
GRANT IMPORTED PRIVILEGES ON DATABASE SNOWFLAKE TO ROLE DATAMIND_ROLE;

-- 9. Cortex access
GRANT DATABASE ROLE SNOWFLAKE.CORTEX_USER TO ROLE DATAMIND_ROLE;

-- 10. Create the cost view AS ACCOUNTADMIN (owns it)
CREATE OR REPLACE VIEW DATAMIND_DB.METADATA.VW_QUERY_COST AS
SELECT
    QUERY_ID,
    LEFT(QUERY_TEXT, 400)                        AS query_text,
    USER_NAME,
    WAREHOUSE_NAME,
    DATABASE_NAME,
    SCHEMA_NAME,
    ROUND(EXECUTION_TIME / 1000, 2)              AS exec_sec,
    ROUND(CREDITS_USED_CLOUD_SERVICES, 6)        AS credits,
    ROUND(CREDITS_USED_CLOUD_SERVICES * 2.0, 4)  AS est_usd,
    ROUND(BYTES_SCANNED / POWER(1024,3), 2)      AS gb_scanned,
    PARTITIONS_SCANNED,
    PARTITIONS_TOTAL,
    ROUND(PARTITIONS_SCANNED * 100.0
          / NULLIF(PARTITIONS_TOTAL, 0), 1)      AS pct_partitions_scanned,
    START_TIME
FROM SNOWFLAKE.ACCOUNT_USAGE.QUERY_HISTORY
WHERE START_TIME       >= DATEADD('day', -7, CURRENT_TIMESTAMP())
  AND EXECUTION_STATUS  = 'SUCCESS'
ORDER BY CREDITS_USED_CLOUD_SERVICES DESC
LIMIT 500;

-- Grant select on the view to the role
GRANT SELECT ON VIEW DATAMIND_DB.METADATA.VW_QUERY_COST TO ROLE DATAMIND_ROLE;

-- 11. Verify
SELECT 'User'      AS object, 'DATAMIND_USER' AS name FROM DUAL
UNION ALL
SELECT 'Role',      'DATAMIND_ROLE'
UNION ALL
SELECT 'Warehouse', 'DATAMIND_WH'
UNION ALL
SELECT 'Database',  'DATAMIND_DB';
